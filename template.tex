$$
\int_0^{\sqrt[3]{\pi}} x^5 \cos{x^3} \, dx

\textbf{2.} Which of the following series converge?
\begin{enumerate}
    \item[(i)] $\sum_{n=1}^\infty \frac{e^{-n^2}}{n^2}$
    \item[(ii)] $\sum_{n=2}^\infty \frac{1}{n (\ln{n})^2}$
\end{enumerate}
(a) (i) only (b) (ii) only (c) (i) and (ii) (d) neither
